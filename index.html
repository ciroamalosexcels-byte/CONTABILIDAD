String(v).trim() : ''; return txt.slice(0, 10); } function fmtHora_(v) { if (v instanceof Date) { return Utilities.formatDate(v, Session.getScriptTimeZone(), 'HH:mm:ss'); } const txt = v ? String(v).trim() : ''; const match = txt.match(/\b\d{2}:\d{2}(:\d{2})?\b/); if (!match) return ''; const hora = match[0]; return hora.length === 5 ? hora + ':00' : hora; } function separarFechaHora_(valor) { const txt = valor ? String(valor).trim().replace('T', ' ') : ''; return { fecha: fmtFecha_(txt), hora: fmtHora_(txt) }; } function unirFechaHora_(fecha, hora) { const f = fmtFecha_(fecha); const h = fmtHora_(hora); if (f && h) return f + ' ' + h; if (f) return f; return ''; } function readState_() { const shM = sheet_(SHEET_MOV); const values = shM.getDataRange().getValues(); const movs = []; if (values.length > 1) { const headers = values[0].map(h => String(h).trim()); // Detecta si la planilla ya tiene columnas nuevas const hasNewCols = headers.includes('hora') && headers.includes('fechaHora'); for (let i = 1; i < values.length; i++) { const r = values[i]; if (r[0] === '' && r[3] === '') continue; let fechaFinal = ''; let vencFinal = ''; let estadoFinal = ''; let fijoFinal = false; if (hasNewCols) { // Nueva estructura: // 0 id, 1 tipo, 2 monto, 3 desc, 4 cat, 5 fecha, 6 hora, 7 fechaHora, 8 venc, 9 estado, 10 fijo const fecha = fmtFecha_(r[5]); const hora = fmtHora_(r[6]); const fechaHora = r[7] ? String(r[7]).trim() : ''; fechaFinal = fechaHora || unirFechaHora_(fecha, hora); vencFinal = fmtFecha_(r[8]); estadoFinal = r[9] || 'pago'; fijoFinal = r[10] === true || r[10] === 'true' || r[10] === 'TRUE'; } else { // Compatibilidad con estructura vieja: // 0 id, 1 tipo, 2 monto, 3 desc, 4 cat, 5 fecha, 6 venc, 7 estado, 8 fijo fechaFinal = String(r[5] || '').trim(); vencFinal = fmtFecha_(r[6]); estadoFinal = r[7] || 'pago'; fijoFinal = r[8] === true || r[8] === 'true' || r[8] === 'TRUE'; } movs.push({ id: String(r[0]), tipo: r[1], monto: Number(r[2]) || 0, desc: r[3], cat: r[4], fecha: fechaFinal, venc: vencFinal, estado: estadoFinal, fijo: fijoFinal }); } } const shC = sheet_(SHEET_CFG); let cfg = {}; const raw = shC.getRange('A1').getValue(); if (raw) { try { cfg = JSON.parse(raw); } catch (e) {} } return { movs: movs, fijos: cfg.fijos || [], cats: cfg.cats || [], goalMin: cfg.goalMin || 1200000, goalMax: cfg.goalMax || 1600000 }; } function writeState_(state) { const shM = sheet_(SHEET_MOV); shM.clearContents(); const rows = [COLS]; (state.movs || []).forEach(function (m) { const partes = separarFechaHora_(m.fecha || ''); const fecha = partes.fecha; const hora = partes.hora; const fechaHora = unirFechaHora_(fecha, hora); rows.push([ m.id || '', m.tipo || '', m.monto || 0, m.desc || '', m.cat || '', "'" + fecha, "'" + hora, "'" + fechaHora, "'" + (m.venc || ''), m.estado || 'pago', m.fijo ? true : false ]); }); shM.getRange(1, 1, rows.length, COLS.length).setValues(rows); // Fuerza formato texto para que Google Sheets no borre la hora if (rows.length > 1) { shM.getRange(2, 6, rows.length - 1, 4).setNumberFormat('@'); } const shC = sheet_(SHEET_CFG); shC.getRange('A1').setValue(JSON.stringify({ fijos: state.fijos || [], cats: state.cats || [], goalMin: state.goalMin, goalMax: state.goalMax })); }
